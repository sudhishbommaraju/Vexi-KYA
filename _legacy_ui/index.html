agent.ts 
import { VexiClient } from '@vexi/node';

// Initialize client
const vexi = new VexiClient({ token: process.env.VEXI_KEY });

// Generate bounded token
const session = await vexi.tokens.create({
  policy: 'pol_aws_deployment_only',
  max_spend: 25000, // $250.00
  expires_in: '1h',
  agent_id: 'ag_deployer_01'
});

// Pass to LLM
await llm.prompt(
  `Deploy architecture. Payment token: ${session.key}`
);
